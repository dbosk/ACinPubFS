% Push Model
\DeclareMathOperator{\PushOp}{\mathcal{P}_n}
\NewDocumentCommand{\Push}{m o}{\ensuremath{%
    \PushOp\left[ #1 \right]%
    \IfValueT{#2}{%
      \!.\mathsf{#2}%
    }
  }}
\NewDocumentCommand{\pnsetup}{m o}{\ensuremath{%
    \Push{#1}[setup]%
    \IfValueT{#2}{%
      \left( #2 \right)%
    }
  }}
\NewDocumentCommand{\pnpub}{m o o}{\ensuremath{%
    \Push{#1}[publish]%
    \IfValueT{#2}{%
      \left( #2, \IfValueTF{#3}{#3}{\cdot} \right)%
    }
  }}
\NewDocumentCommand{\pnget}{m o}{\ensuremath{%
    \Push{#1}[fetch]%

  }}
